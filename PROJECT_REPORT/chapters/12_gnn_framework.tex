\chapter{Graph Neural Network Framework}
\label{chap:gnn}

This chapter presents NOMA-GNN, our proposed GraphSAGE-based framework for learning optimal user pairing and power allocation.

\section{Motivation for Graph-Based Learning}

Traditional optimization suffers from:
\begin{itemize}
    \item $O(N^3)$ complexity prohibits real-time use for large $N$
    \item Requires re-solving for every channel realization
    \item Separate pairing and power allocation stages yield suboptimal joint solutions
\end{itemize}

\textbf{Insight:} User pairing naturally forms a graph structure where:
\begin{itemize}
    \item Nodes represent users with channel/spatial features
    \item Edges represent candidate pairings with rate/power attributes
    \item Optimal matching is a subset of edges
\end{itemize}

Graph Neural Networks can learn this structure-to-matching mapping from data.

\section{Problem as Graph Learning}

\subsection{Graph Construction}

Given $N$ users with CSI, construct undirected graph $\mathcal{G} = (\mathcal{V}, \mathcal{E})$:

\textbf{Nodes} $\mathcal{V} = \{1, \ldots, N\}$ with feature vectors:
\begin{equation}
\mathbf{x}_i = \begin{bmatrix}
|h_i|^2_{\text{dB}} \\
d_i \\
\theta_i \\
\text{SNR}_i \\
PL_{i,\text{dB}}
\end{bmatrix} \in \mathbb{R}^5
\end{equation}

\textbf{Edges} $\mathcal{E}$ include $(i,j)$ if:
\begin{enumerate}
    \item \textbf{SIC Feasibility:} $|h_j|^2 / |h_i|^2 \geq \gamma_{th}$ (assuming $j > i$ in channel gain)
    \item \textbf{Angular Separation:} $|\theta_i - \theta_j| \geq \Delta\theta_{min} = 25°$
    \item \textbf{Ordering:} $i < j$ (to avoid duplicates)
\end{enumerate}

This yields a \emph{sparse bipartite-like} graph where weak users connect only to compatible strong users.

\subsection{Learning Objectives}

For each edge $(i,j) \in \mathcal{E}$, predict three quantities:

\begin{enumerate}
    \item \textbf{Pairing Probability} $p_{ij} \in [0,1]$: Binary classification
    \begin{itemize}
        \item Target: $y_{ij} = 1$ if $(i,j)$ in optimal Blossom matching, else 0
        \item Loss: Binary Cross-Entropy (BCE)
    \end{itemize}
    
    \item \textbf{Sum-Rate} $\hat{R}_{ij}$: Regression
    \begin{itemize}
        \item Target: $R_{ij} = R_i + R_j$ computed via NOMA rate formula
        \item Loss: Mean Absolute Error (MAE)
    \end{itemize}
    
    \item \textbf{Power Allocation} $\hat{\alpha}_{ij} \in (0,1)$: Regression
    \begin{itemize}
        \item Target: $\alpha_{ij}^*$ from bisection search
        \item Loss: MAE
    \end{itemize}
\end{enumerate}

\textbf{Rationale for Multi-Task:}
\begin{itemize}
    \item Pairing decisions depend on achievable rates (coupling)
    \item Power splits affect rates (feedback)
    \item Shared representations improve generalization
\end{itemize}

\section{GraphSAGE Architecture}

\subsection{Why GraphSAGE?}

Among GNN variants, GraphSAGE offers:
\begin{enumerate}
    \item \textbf{Inductive Learning:} Generalizes to unseen graph sizes (varying $N$)
    \item \textbf{Sampling Efficiency:} Neighborhood sampling scales to large graphs
    \item \textbf{Expressive Aggregation:} Mean/max aggregators capture diverse patterns
\end{enumerate}

\subsection{Encoder: Node Embedding}

Transform raw node features into high-dimensional latent representations through $L=3$ message-passing layers.

\textbf{Layer 0 (Input Projection):}
\begin{equation}
\mathbf{h}_i^{(0)} = \text{ReLU}\left( \mathbf{W}_{in} \mathbf{x}_i + \mathbf{b}_{in} \right) \in \mathbb{R}^{128}
\end{equation}

where $\mathbf{W}_{in} \in \mathbb{R}^{128 \times 5}$ projects 5D features to 128D.

\textbf{Layers 1-3 (Message Passing):}
\begin{equation}
\mathbf{h}_i^{(\ell)} = \text{ReLU}\left( \mathbf{W}_{\ell} \cdot \text{AGG}\left( \{\mathbf{h}_j^{(\ell-1)} : j \in \mathcal{N}(i)\} \right) + \mathbf{b}_{\ell} \right)
\end{equation}

where:
\begin{itemize}
    \item $\mathcal{N}(i)$ is the neighborhood of node $i$ in the candidate graph
    \item $\text{AGG}(\cdot)$ is the \textbf{mean aggregator}:
    \begin{equation}
    \text{AGG}(\{\mathbf{h}_j\}) = \frac{1}{|\mathcal{N}(i)|} \sum_{j \in \mathcal{N}(i)} \mathbf{h}_j
    \end{equation}
    \item $\mathbf{W}_{\ell} \in \mathbb{R}^{128 \times 128}$ are learnable weight matrices
    \item ReLU activation: $\text{ReLU}(x) = \max(0, x)$
\end{itemize}

\textbf{Final Node Embedding:}
\begin{equation}
\mathbf{z}_i = \mathbf{h}_i^{(3)} \in \mathbb{R}^{128}
\end{equation}

\textbf{Receptive Field:}
\begin{itemize}
    \item Layer 1: 1-hop neighbors
    \item Layer 2: 2-hop neighbors
    \item Layer 3: 3-hop neighbors
\end{itemize}

This enables each node to aggregate information from distant nodes, capturing global graph structure.

\subsection{Decoder: Multi-Task Edge Prediction}

For each edge $(i,j) \in \mathcal{E}$, concatenate learned node embeddings:
\begin{equation}
\mathbf{e}_{ij} = [\mathbf{z}_i \, \| \, \mathbf{z}_j] \in \mathbb{R}^{256}
\end{equation}

Pass through three specialized Multi-Layer Perceptron (MLP) heads:

\subsubsection{Pairing Classifier}

\begin{equation}
\begin{aligned}
\mathbf{h}_{ij}^{(1)} &= \text{ReLU}(\mathbf{W}_p^{(1)} \mathbf{e}_{ij} + \mathbf{b}_p^{(1)}) \in \mathbb{R}^{128} \\
\mathbf{h}_{ij}^{(1)} &= \text{Dropout}(\mathbf{h}_{ij}^{(1)}, p=0.3) \\
\mathbf{h}_{ij}^{(2)} &= \text{ReLU}(\mathbf{W}_p^{(2)} \mathbf{h}_{ij}^{(1)} + \mathbf{b}_p^{(2)}) \in \mathbb{R}^{64} \\
\mathbf{h}_{ij}^{(2)} &= \text{Dropout}(\mathbf{h}_{ij}^{(2)}, p=0.3) \\
p_{ij} &= \text{Sigmoid}(\mathbf{w}_p^{(3)} \mathbf{h}_{ij}^{(2)} + b_p^{(3)}) \in (0,1)
\end{aligned}
\end{equation}

Architecture: 256 → 128 → 64 → 1

\subsubsection{Sum-Rate Regressor}

\begin{equation}
\begin{aligned}
\mathbf{h}_{ij}^{(1)} &= \text{ReLU}(\mathbf{W}_r^{(1)} \mathbf{e}_{ij} + \mathbf{b}_r^{(1)}) \in \mathbb{R}^{128} \\
\mathbf{h}_{ij}^{(2)} &= \text{ReLU}(\mathbf{W}_r^{(2)} \mathbf{h}_{ij}^{(1)} + \mathbf{b}_r^{(2)}) \in \mathbb{R}^{64} \\
\hat{R}_{ij} &= \text{ReLU}(\mathbf{w}_r^{(3)} \mathbf{h}_{ij}^{(2)} + b_r^{(3)}) \in \mathbb{R}_+
\end{aligned}
\end{equation}

Final ReLU ensures non-negative rates.

\subsubsection{Power Allocation Regressor}

\begin{equation}
\begin{aligned}
\mathbf{h}_{ij}^{(1)} &= \text{ReLU}(\mathbf{W}_a^{(1)} \mathbf{e}_{ij} + \mathbf{b}_a^{(1)}) \in \mathbb{R}^{128} \\
\mathbf{h}_{ij}^{(2)} &= \text{ReLU}(\mathbf{W}_a^{(2)} \mathbf{h}_{ij}^{(1)} + \mathbf{b}_a^{(2)}) \in \mathbb{R}^{64} \\
\hat{\alpha}_{ij} &= \text{Sigmoid}(\mathbf{w}_a^{(3)} \mathbf{h}_{ij}^{(2)} + b_a^{(3)}) \in (0,1)
\end{aligned}
\end{equation}

Sigmoid constrains output to valid power split range.

\section{Training Procedure}

\subsection{Dataset Generation}

\textbf{Ground Truth Labels:}
\begin{enumerate}
    \item Generate 50,000 user scenarios with $N \in \{4,6,8,10,12\}$
    \item For each scenario:
    \begin{itemize}
        \item Run Blossom matching to get optimal pairs
        \item Compute sum-rates via NOMA formula
        \item Find optimal $\alpha$ via bisection search
    \end{itemize}
    \item Construct graph with candidate edges
    \item Label edges: $y_{ij} = 1$ if in optimal matching, else 0
\end{enumerate}

\textbf{Data Split:}
\begin{itemize}
    \item Training: 35,000 scenarios (70\%)
    \item Validation: 7,500 scenarios (15\%)
    \item Test: 7,500 scenarios (15\%)
\end{itemize}

\subsection{Hard Negative Sampling}

Challenge: Class imbalance—most candidate edges are not in optimal matching (negative class dominates).

\textbf{Solution:} During training, sample:
\begin{itemize}
    \item \textbf{All positive edges:} In optimal matching
    \item \textbf{Hard negatives:} Top-K candidate edges by predicted pairing score but not in optimal matching
    \item \textbf{Random negatives:} Uniformly sampled from remaining candidates
\end{itemize}

Ratio: 1 positive : 2 hard negatives : 1 random negative.

\textbf{Implementation:}
\begin{lstlisting}[language=Python, caption={Hard Negative Sampling}]
def sample_edges(graph_data, model, k_hard=2):
    """Sample positive, hard negative, and random negative edges."""
    # Get all candidate edges
    edge_index = graph_data.edge_index
    labels = graph_data.edge_labels
    
    # Positive edges
    pos_mask = (labels == 1)
    pos_edges = edge_index[:, pos_mask]
    
    # Negative edges
    neg_mask = ~pos_mask
    neg_edges = edge_index[:, neg_mask]
    
    # Predict scores for negative edges
    with torch.no_grad():
        neg_scores = model.predict_pairing(graph_data, neg_edges)
    
    # Hard negatives: top-K by score
    num_hard = k_hard * pos_edges.shape[1]
    hard_idx = torch.topk(neg_scores, num_hard).indices
    hard_neg_edges = neg_edges[:, hard_idx]
    
    # Random negatives
    remaining_idx = torch.tensor([i for i in range(neg_edges.shape[1])
                                 if i not in hard_idx])
    num_random = pos_edges.shape[1]
    random_idx = remaining_idx[torch.randperm(len(remaining_idx))[:num_random]]
    random_neg_edges = neg_edges[:, random_idx]
    
    # Combine
    sampled_edges = torch.cat([pos_edges, hard_neg_edges, random_neg_edges], dim=1)
    sampled_labels = torch.cat([
        torch.ones(pos_edges.shape[1]),
        torch.zeros(hard_neg_edges.shape[1] + random_neg_edges.shape[1])
    ])
    
    return sampled_edges, sampled_labels
\end{lstlisting}

\subsection{Multi-Task Loss Function}

\textbf{Total Loss:}
\begin{equation}
\mathcal{L}_{total} = \lambda_p \mathcal{L}_{pair} + \lambda_r \mathcal{L}_{rate} + \lambda_a \mathcal{L}_{power}
\end{equation}

\textbf{Pairing Loss (BCE):}
\begin{equation}
\mathcal{L}_{pair} = -\frac{1}{|\mathcal{E}|} \sum_{(i,j) \in \mathcal{E}} \left[ y_{ij} \log p_{ij} + (1-y_{ij}) \log(1-p_{ij}) \right]
\end{equation}

\textbf{Sum-Rate Loss (MAE on positive edges):}
\begin{equation}
\mathcal{L}_{rate} = \frac{1}{|\mathcal{E}_+|} \sum_{(i,j) \in \mathcal{E}_+} |\hat{R}_{ij} - R_{ij}|
\end{equation}

where $\mathcal{E}_+ = \{(i,j) : y_{ij} = 1\}$ (only compute for actual pairs).

\textbf{Power Allocation Loss (MAE on positive edges):}
\begin{equation}
\mathcal{L}_{power} = \frac{1}{|\mathcal{E}_+|} \sum_{(i,j) \in \mathcal{E}_+} |\hat{\alpha}_{ij} - \alpha_{ij}^*|
\end{equation}

\textbf{Loss Weights:}
\begin{itemize}
    \item $\lambda_p = 1.0$ (pairing is primary task)
    \item $\lambda_r = 0.5$ (sum-rate provides supervision signal)
    \item $\lambda_a = 0.3$ (power allocation fine-tunes)
\end{itemize}

Weights determined via validation set hyperparameter search.

\subsection{Optimization}

\textbf{Optimizer:} Adam with:
\begin{itemize}
    \item Learning rate: $lr = 10^{-3}$
    \item $\beta_1 = 0.9$, $\beta_2 = 0.999$
    \item Weight decay: $\lambda_{wd} = 10^{-4}$ (L2 regularization)
\end{itemize}

\textbf{Learning Rate Scheduler:} ReduceLROnPlateau
\begin{itemize}
    \item Monitor: Validation AUC
    \item Factor: 0.5 (halve LR on plateau)
    \item Patience: 10 epochs
    \item Minimum LR: $10^{-6}$
\end{itemize}

\textbf{Early Stopping:}
\begin{itemize}
    \item Monitor: Validation AUC
    \item Patience: 25 epochs
    \item Restore best model weights
\end{itemize}

\textbf{Training Loop:}
\begin{lstlisting}[language=Python, caption={Training Loop}]
def train_epoch(model, train_loader, optimizer):
    model.train()
    total_loss = 0
    
    for batch in train_loader:
        optimizer.zero_grad()
        
        # Sample edges with hard negatives
        edges, labels = sample_edges(batch, model)
        
        # Forward pass
        p_pair, R_sum, alpha = model(batch, edges)
        
        # Compute losses
        loss_pair = F.binary_cross_entropy(p_pair, labels)
        
        pos_mask = (labels == 1)
        if pos_mask.sum() > 0:
            loss_rate = F.l1_loss(R_sum[pos_mask], 
                                  batch.rate_targets[pos_mask])
            loss_power = F.l1_loss(alpha[pos_mask],
                                   batch.alpha_targets[pos_mask])
        else:
            loss_rate = loss_power = 0.0
        
        # Total loss
        loss = (1.0 * loss_pair + 
                0.5 * loss_rate + 
                0.3 * loss_power)
        
        # Backward pass
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
        optimizer.step()
        
        total_loss += loss.item()
    
    return total_loss / len(train_loader)
\end{lstlisting}

\section{Inference Pipeline}

\subsection{Forward Pass}

Given new user scenario:
\begin{enumerate}
    \item Construct candidate graph $\mathcal{G}$ with SIC/angular constraints
    \item Compute node embeddings via GraphSAGE encoder
    \item Predict $(p_{ij}, \hat{R}_{ij}, \hat{\alpha}_{ij})$ for all edges
\end{enumerate}

\subsection{Greedy Matching}

\begin{algorithm}[H]
\caption{Greedy Matching via Predicted Rates}
\label{alg:greedy_gnn}
\begin{algorithmic}[1]
\Require Edge predictions $(p_{ij}, \hat{R}_{ij})$ for all $(i,j) \in \mathcal{E}$
\Ensure Pairing $\mathcal{M}$

\State Sort edges by predicted sum-rate: $\mathcal{E}_{sorted} \gets \text{sort}(\mathcal{E}, \text{key}=\hat{R}_{ij}, \text{reverse}=\text{True})$
\State $\mathcal{M} \gets \emptyset$ \Comment{Matched pairs}
\State $\mathcal{U} \gets \{1, \ldots, N\}$ \Comment{Unmatched users}

\For{$(i,j) \in \mathcal{E}_{sorted}$}
    \If{$i \in \mathcal{U}$ \textbf{and} $j \in \mathcal{U}$}
        \State $\mathcal{M} \gets \mathcal{M} \cup \{(i,j)\}$
        \State $\mathcal{U} \gets \mathcal{U} \setminus \{i, j\}$
    \EndIf
    
    \If{$|\mathcal{U}| \leq 1$}
        \State \textbf{break} \Comment{No more pairs possible}
    \EndIf
\EndFor

\State \Return $\mathcal{M}$
\end{algorithmic}
\end{algorithm}

\textbf{Complexity:}
\begin{itemize}
    \item Sorting: $O(|\mathcal{E}| \log |\mathcal{E}|)$ where $|\mathcal{E}| \ll N^2$ due to constraints
    \item Greedy selection: $O(|\mathcal{E}|)$
    \item Total: $O(|\mathcal{E}| \log |\mathcal{E}|) = O(N \log N)$ in practice
\end{itemize}

\subsection{SIC Verification}

For each selected pair $(i,j) \in \mathcal{M}$, verify predicted power allocation satisfies SIC:
\begin{equation}
\text{SINR}_{j \to i}(\hat{\alpha}_{ij}) = \frac{\hat{\alpha}_{ij} P |h_j|^2}{(1-\hat{\alpha}_{ij}) P |h_j|^2 + \sigma^2} \geq \gamma_{th}
\end{equation}

If violated, adjust $\hat{\alpha}_{ij}$ to minimum feasible value:
\begin{equation}
\alpha_{ij,\text{adjusted}} = \max\left(\hat{\alpha}_{ij}, \frac{\gamma_{th} \sigma^2}{\gamma_{th} P |h_j|^2 - (1-\gamma_{th}) P |h_j|^2}\right)
\end{equation}

In practice, 98.7\% of predictions satisfy SIC without adjustment.

\subsection{Complete Inference}

\begin{lstlisting}[language=Python, caption={NOMA-GNN Inference}]
def infer_pairing(model, users_data):
    """
    Perform real-time pairing inference.
    
    Args:
        model: Trained NOMA-GNN
        users_data: Dict with keys 'h_values', 'angles', etc.
    
    Returns:
        List of paired (user_i, user_j, alpha_ij)
    """
    # Build candidate graph
    graph = construct_graph(users_data, 
                           sic_threshold_db=8.0,
                           min_angle_deg=25.0)
    
    # Forward pass
    model.eval()
    with torch.no_grad():
        p_pair, R_sum, alpha = model(graph)
    
    # Greedy matching on predicted rates
    edges = graph.edge_index.T.numpy()
    rates = R_sum.numpy()
    alphas = alpha.numpy()
    
    # Sort by rate
    sorted_idx = np.argsort(-rates)  # descending
    
    matched_users = set()
    pairs = []
    
    for idx in sorted_idx:
        i, j = edges[idx]
        if i not in matched_users and j not in matched_users:
            # Verify SIC with predicted alpha
            if verify_sic(users_data['h_values'][i],
                         users_data['h_values'][j],
                         alphas[idx]):
                pairs.append((i, j, alphas[idx]))
                matched_users.update([i, j])
        
        if len(matched_users) >= len(users_data['h_values']):
            break
    
    return pairs
\end{lstlisting}

\section{Model Interpretability}

\subsection{Embedding Visualization}

We apply t-SNE to learned node embeddings $\{\mathbf{z}_i\}$ to visualize clustering:

\begin{itemize}
    \item \textbf{Observation:} Nodes naturally cluster by channel gain magnitude
    \item Weak users form one cluster, strong users another
    \item Spatial position (angle, distance) creates sub-clusters within each group
\end{itemize}

This indicates the model learns strong-weak complementarity for NOMA pairing.

\subsection{Feature Importance}

Ablation by removing features:
\begin{itemize}
    \item Removing angular features: AUC drops 4.2\% (88.3\% vs 92.5\%)
    \item Removing distance: AUC drops 1.8\% (90.7\%)
    \item Removing path loss: AUC drops 0.6\% (91.9\%)
\end{itemize}

Conclusion: Angular separation is critical for effective pairing.

\subsection{Attention Mechanisms (Future Work)}

Current mean aggregation treats neighbors equally. Future versions could incorporate Graph Attention Networks (GATs) to learn adaptive neighbor weights, improving interpretability.
