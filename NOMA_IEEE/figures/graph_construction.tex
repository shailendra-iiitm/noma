% Constraint boxes
\node[draw, rectangle, thick, fill=yellow!20, text width=32mm, font=\tiny, align=left] at (-4.5, 0) {

$\bullet$ $|\theta_i - \theta_j| \geq 25^\circ$\\
$\bullet$ $|h_i|^2 < |h_j|^2$
};
