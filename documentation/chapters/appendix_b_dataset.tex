\section{Normalization Scheme}

\subsection{Input Features}

\begin{table}[H]
\centering
\begin{tabular}{ll}
\toprule
\textbf{Feature} & \textbf{Normalization} \\
\midrule

$x, y$ & StandardScaler: $z = (x - \mu) / \sigma$ \\

\section{Data Splits}

\subsection{Split Strategy}

\begin{itemize}
    \item \textbf{Random scenario split:} Scenarios randomly assigned to train/val/test
    \item \textbf{Stratified by N:} Ensure balanced user counts across splits
    \item \textbf{No data leakage:} All users/edges from same scenario in same split
\end{itemize}

\section{Data Generation Pipeline}

\begin{enumerate}
    \item \textbf{User Placement:} Uniform random in circular cell (radius 5000 m)
    \item \textbf{Channel Generation:} 3GPP UMa model (LOS/NLOS, shadowing, fading)
    \item \textbf{Optimal Pairing:} Blossom maximum-weight matching
    \item \textbf{Power Optimization:} Bisection search for each pair
    \item \textbf{Negative Sampling:} Hard negatives via top-K non-paired edges
    \item \textbf{Serialization:} Save to CSV with normalized features
\end{enumerate}

\section{Data Validation}

\subsection{Validation Checks}

\begin{itemize}
    \item \textbf{Channel consistency:} All $h_i > 0$
    \item \textbf{Distance bounds:} $10 \leq d_i \leq 5000$ m
    \item \textbf{Power constraint:} $P_i + P_j = 1$ W for all pairs
    \item \textbf{SIC feasibility:} $|h_i - h_j| \geq 8$ dB for all positive pairs
    \item \textbf{No duplicate scenarios:} All scenario IDs unique
    \item \textbf{Balanced splits:} User counts uniformly distributed across splits
\end{itemize}
